#+BEGIN_SRC emacs-lisp
  (setq dired-listing-switches "-al")

Add handy keybind to WDired.
#+BEGIN_SRC emacs-lisp
  (add-hook 'dired-mode-hook
            (lambda ()
              (local-set-key
               (kbd "W")
               'wdired-change-to-wdired-mode)))
#+END_SRC
